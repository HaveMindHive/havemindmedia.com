2.3 Arduino Code

// Training Device Firmware
// Generates κ-shadow (28.65 Hz) vibration

const int MOTOR_L = 9;
const int MOTOR_R = 10;
const int POT_PIN = A0;

const float KAPPA_SHADOW = 28.65; // Hz
const int PERIOD_US = 1000000 / KAPPA_SHADOW; // ~34,900 µs

void setup() {
    pinMode(MOTOR_L, OUTPUT);
    pinMode(MOTOR_R, OUTPUT);
    Serial.begin(115200);
}

void loop() {
    // Read potentiometer for balance
    int pot = analogRead(POT_PIN);
    float balance = (pot / 1023.0) * 2.0 - 1.0; // -1 to +1

    // Calculate motor intensities
    int left_intensity = constrain(128 + balance * 127, 0, 255);
    int right_intensity = constrain(128 - balance * 127, 0, 255);

    // Generate κ-shadow frequency
    analogWrite(MOTOR_L, left_intensity);
    analogWrite(MOTOR_R, right_intensity);

    delayMicroseconds(PERIOD_US / 2);

    analogWrite(MOTOR_L, 0);
    analogWrite(MOTOR_R, 0);

    delayMicroseconds(PERIOD_US / 2);
}

# node_firmware.py - MicroPython for Pico

import machine
import time
from drv2605 import DRV2605
from neopixel import NeoPixel

# Configuration
NODE_ID = 0  # 0=NW, 1=NE, 2=SW, 3=SE
KAPPA_SHADOW = 28.65

# Pins
FSR_PIN = machine.ADC(26)
LED_PIN = machine.Pin(16, machine.Pin.OUT)
I2C_SDA = machine.Pin(0)
I2C_SCL = machine.Pin(1)

# Initialize haptic driver
i2c = machine.I2C(0, sda=I2C_SDA, scl=I2C_SCL)
haptic = DRV2605(i2c)

# Initialize LED
led = NeoPixel(LED_PIN, 1)

# State
tau = 0.0
intensity = 0

def set_tau(value):
    """Set torsion value (-1 to +1)."""
    global tau, intensity
    tau = max(-1.0, min(1.0, value))

    # Map to intensity (0-255)
    intensity = int(abs(tau) * 255)

    # Set LED color based on sign
    if tau > 0:
        led[0] = (0, intensity, intensity)  # Cyan for positive
    else:
        led[0] = (intensity, intensity // 2, 0)  # Orange for negative
    led.write()

def read_force():
    """Read FSR sensor, return normalized 0-1."""
    raw = FSR_PIN.read_u16()
    return raw / 65535.0

def vibrate():
    """Generate κ-shadow vibration at current intensity."""
    if intensity > 0:
        haptic.set_waveform(0, 1)  # Strong click
        haptic.go()

# Main loop
while True:
    # Check for serial commands from hub
    if machine.stdin.any():
        cmd = machine.stdin.readline().strip()
        if cmd.startswith('T'):
            # Tau command: T0.5 or T-0.3
            set_tau(float(cmd[1:]))

    # Send force reading to hub
    force = read_force()
    print(f"F{NODE_ID}:{force:.4f}")

    # Vibrate at κ-shadow frequency
    vibrate()
    time.sleep_us(int(1000000 / KAPPA_SHADOW / 2))

3.5 Hub Software Integration

# hub_interface.py - For Raspberry Pi 4

import serial
import threading
from sphere_os import SphereOS1729

class HardwareHub:
    def __init__(self):
        self.sphere = SphereOS1729()
        self.nodes = {}

        # Connect to node Picos
        ports = ['/dev/ttyACM0', '/dev/ttyACM1',
                 '/dev/ttyACM2', '/dev/ttyACM3']
        names = ['NW', 'NE', 'SW', 'SE']

        for port, name in zip(ports, names):
            try:
                self.nodes[name] = serial.Serial(port, 115200)
            except:
                print(f"Warning: Could not connect to {name}")

    def send_torsion(self):
        """Send current torsion values to nodes."""
        for name, ser in self.nodes.items():
            tau = self.sphere.field.nodes[name].tau
            ser.write(f"T{tau:.4f}\n".encode())

    def read_forces(self):
        """Read force values from nodes."""
        forces = {}
        for name, ser in self.nodes.items():
            if ser.in_waiting:
                line = ser.readline().decode().strip()
                if line.startswith('F'):
                    _, value = line.split(':')
                    forces[name] = float(value)
        return forces

    def tick(self):
        """One hardware tick cycle."""
        # Read sensor input
        forces = self.read_forces()

        # Apply to simulation
        for name, force in forces.items():
            # Convert force to intent
            intent = (force - 0.5) * 2  # Center at 0
            self.sphere.field.apply_input(name, intent)

        # Run simulation tick
        self.sphere.tick()

        # Send output to hardware
        self.send_torsion()

    def run(self):
        """Main loop."""
        print("1729 Sphere OS Hardware Hub")
        print("Press Ctrl+C to stop")

        try:
            while True:
                self.tick()
                time.sleep(1.0 / KAPPA_SHADOW)
        except KeyboardInterrupt:
            print("\nShutting down...")

if __name__ == '__main__':
    hub = HardwareHub()
    hub.run()

4.3 EEG Integration

# eeg_interface.py - OpenBCI integration

import brainflow
from brainflow.board_shim import BoardShim, BrainFlowInputParams

class EEGInterface:
    def __init__(self):
        params = BrainFlowInputParams()
        params.serial_port = '/dev/ttyUSB0'  # OpenBCI dongle

        self.board = BoardShim(
            brainflow.BoardIds.CYTON_DAISY_BOARD.value,
            params
        )
        self.board.prepare_session()
        self.board.start_stream()

        self.beta_power = 0.0
        self.kappa_sync = 0.0

    def update(self):
        """Read EEG data and compute metrics."""
        data = self.board.get_current_board_data(256)  # 1 second

        # Extract beta band (13-30 Hz) power
        for channel in range(8):
            psd = brainflow.DataFilter.get_psd(
                data[channel],
                256,  # Sample rate
                brainflow.WindowTypes.HANNING.value
            )
            # Find power in 13-30 Hz range
            beta_mask = (psd[1] >= 13) & (psd[1] <= 30)
            self.beta_power = np.mean(psd[0][beta_mask])

        # Check for κ-shadow (28.65 Hz) synchronization
        kappa_mask = (psd[1] >= 28) & (psd[1] <= 30)
        kappa_power = np.mean(psd[0][kappa_mask])
        self.kappa_sync = kappa_power / (self.beta_power + 0.001)

    def get_consciousness_estimate(self):
        """Return estimated consciousness alignment (0-1)."""
        # Higher κ-sync relative to general beta = better alignment
        return min(1.0, self.kappa_sync / 2.0)

6. Testing Protocol

6.1 Hardware Validation

  1. Vibration Frequency Test
    • Measure actual motor frequency with oscilloscope
    • Verify 28.65 Hz ±0.5 Hz
  2. Sensor Calibration
    • Apply known forces, record ADC values
    • Create calibration curve for each sensor
  3. Latency Test
    • Measure input-to-output delay
    • Target: <20ms total latency
  4. Cross-talk Test
    • Vibrate one node, check others don't false-trigger
    • Should be <5% cross-talk
