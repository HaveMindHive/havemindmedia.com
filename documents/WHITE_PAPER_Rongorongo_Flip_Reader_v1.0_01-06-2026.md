class HealingEngine:
    def __init__(self):
        self.q_cycle_position = 0
        self.flip_times = []
        self.session_start = None

    def start_session(self):
        """Begin a healing reading session."""
        self.session_start = datetime.now()
        self.q_cycle_position = 0
        self.flip_times = []

    def record_flip(self):
        """Record a flip event for rhythm analysis."""
        self.flip_times.append(datetime.now())
        self.q_cycle_position = (self.q_cycle_position + 1) % 4

    def get_rhythm_analysis(self) -> RhythmAnalysis:
        """
        Analyze the rhythm of flipping for therapeutic consistency.

        Optimal rhythm: ~4-6 seconds per flip (like EMDR bilateral timing)
        """
        if len(self.flip_times) < 2:
            return RhythmAnalysis(intervals=[], average=0, consistency=0)

        intervals = []
        for i in range(1, len(self.flip_times)):
            delta = (self.flip_times[i] - self.flip_times[i-1]).total_seconds()
            intervals.append(delta)

        average = sum(intervals) / len(intervals)
        variance = sum((x - average) ** 2 for x in intervals) / len(intervals)
        consistency = 1 / (1 + variance)  # Higher = more consistent

        return RhythmAnalysis(
            intervals=intervals,
            average=average,
            consistency=consistency,
            optimal_range=(4.0, 6.0)
        )

    def get_q_cycle_completion(self) -> float:
        """Return percentage through current Q-cycle (0-100%)."""
        return (self.q_cycle_position / 4) * 100

    def get_healing_metrics(self) -> HealingMetrics:
        """Compile overall healing session metrics."""
        rhythm = self.get_rhythm_analysis()
        duration = (datetime.now() - self.session_start).total_seconds() if self.session_start else 0
        cycles_completed = len(self.flip_times) // 4

        return HealingMetrics(
            duration_seconds=duration,
            total_flips=len(self.flip_times),
            cycles_completed=cycles_completed,
            rhythm_consistency=rhythm.consistency,
            average_flip_interval=rhythm.average
        )

/**
 * Rongorongo Tablet 3D Visualization

class TabletVisualization {
  constructor(containerId) {
    this.container = document.getElementById(containerId);
    this.scene = new THREE.Scene();
    this.camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
    this.renderer = new THREE.WebGLRenderer({ antialias: true });

    this.tablet = null;
    this.currentRotation = 0;
    this.isFlipping = false;

    this.init();
  }

  init() {
    this.renderer.setSize(this.container.clientWidth, this.container.clientHeight);
    this.container.appendChild(this.renderer.domElement);

    // Lighting
    const ambientLight = new THREE.AmbientLight(0x404040, 0.5);
    this.scene.add(ambientLight);

    const directionalLight = new THREE.DirectionalLight(0xffffff, 0.8);
    directionalLight.position.set(5, 5, 5);
    this.scene.add(directionalLight);

    // Create tablet
    this.createTablet();

    // Camera position
    this.camera.position.z = 5;

    // Start animation loop
    this.animate();

    // Add interaction handlers
    this.setupInteraction();
  }

  createTablet() {
    // Tablet geometry (rectangular wooden tablet)
    const geometry = new THREE.BoxGeometry(4, 2, 0.2);

    // Wood texture
    const textureLoader = new THREE.TextureLoader();
    const woodTexture = textureLoader.load('/textures/wood_dark.jpg');

    const material = new THREE.MeshPhongMaterial({
      map: woodTexture,
      color: 0x8B4513,  // Saddle brown
      shininess: 30
    });

    this.tablet = new THREE.Mesh(geometry, material);
    this.scene.add(this.tablet);

    // Add glyph lines as texture overlay
    this.addGlyphLines();
  }

  addGlyphLines() {
    // Create canvas for glyph rendering
    const canvas = document.createElement('canvas');
    canvas.width = 1024;
    canvas.height = 512;
    const ctx = canvas.getContext('2d');

    // Draw glyphs on canvas (placeholder)
    ctx.fillStyle = '#c9a227';  // Gold color
    ctx.font = '24px serif';

    // Draw sample glyph lines
    for (let line = 0; line < 5; line++) {
      const y = 80 + line * 80;
      ctx.fillText('═══════════════════════════════════', 50, y);
    }

    // Create texture from canvas
    const glyphTexture = new THREE.CanvasTexture(canvas);

    // Create overlay plane
    const overlayGeometry = new THREE.PlaneGeometry(3.8, 1.8);
    const overlayMaterial = new THREE.MeshBasicMaterial({
      map: glyphTexture,
      transparent: true,
      opacity: 0.9
    });

    const overlay = new THREE.Mesh(overlayGeometry, overlayMaterial);
    overlay.position.z = 0.11;  // Slightly above tablet surface
    this.tablet.add(overlay);
  }

  flip() {
    if (this.isFlipping) return;

    this.isFlipping = true;
    const targetRotation = this.currentRotation + Math.PI;  // 180 degrees

    const animate = () => {
      if (this.currentRotation < targetRotation) {
        this.currentRotation += 0.1;  // Rotation speed
        this.tablet.rotation.x = this.currentRotation;
        requestAnimationFrame(animate);
      } else {
        this.currentRotation = targetRotation;
        this.tablet.rotation.x = this.currentRotation;
        this.isFlipping = false;

        // Dispatch flip event
        this.container.dispatchEvent(new CustomEvent('tabletFlipped', {
          detail: { rotations: this.currentRotation / Math.PI }
        }));
      }
    };

    animate();
  }

  setupInteraction() {
    // Keyboard: Spacebar to flip
    document.addEventListener('keydown', (e) => {
      if (e.code === 'Space') {
        e.preventDefault();
        this.flip();
      }
    });

    // Touch: Swipe to flip
    let touchStartY = 0;
    this.container.addEventListener('touchstart', (e) => {
      touchStartY = e.touches[0].clientY;
    });

    this.container.addEventListener('touchend', (e) => {
      const touchEndY = e.changedTouches[0].clientY;
      const deltaY = touchStartY - touchEndY;

      if (Math.abs(deltaY) > 50) {  // Minimum swipe distance
        this.flip();
      }
    });

    // Mouse: Click edges to flip
    this.container.addEventListener('click', (e) => {
      const rect = this.container.getBoundingClientRect();
      const y = e.clientY - rect.top;
      const height = rect.height;

      // Click top or bottom 20% to flip
      if (y < height * 0.2 || y > height * 0.8) {
        this.flip();
      }
    });
  }

  animate() {
    requestAnimationFrame(() => this.animate());

    // Subtle floating animation when not flipping
    if (!this.isFlipping) {
      this.tablet.rotation.y = Math.sin(Date.now() * 0.001) * 0.05;
    }

    this.renderer.render(this.scene, this.camera);
  }
}
