Full Usage

./compile-book.sh [options]

# Options:
  -s, --source DIR      Source directory with .md files (required)
  -o, --output DIR      Output directory (default: ./output)
  -t, --title TITLE     Book title
  -a, --author NAME     Author name (default: Jason Ray)
  -f, --format FMT      Output: pdf, epub, html, or all (default: all)
  -h, --help            Show help

# Example: Compile Genesis of Scalar Dimensionality
./compile-book.sh \
  -s "./GENESIS_SCALAR_DIMENSIONALITY" \
  -o "./genesis-output" \
  -t "Genesis of Scalar Dimensionality" \
  -a "Jason Ray" \
  -f all
